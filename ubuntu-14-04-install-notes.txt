*****Ubuntu 14.04 Fresh Install*****

// Install VIM
sudo apt-get install vim
// Get LAMP Stack in one line
sudo apt-get update
sudo apt-get install lamp-server^

// Install MyPhpAdmin
sudo apt-get update
sudo apt-get install phpmyadmin

// Edit your apache config to point to phpmyadmin apache config
sudo nano /etc/apache2/apache2.conf
 
// Add this line 
Include /etc/phpmyadmin/apache.conf

// Restart server
sudo service apache2 restart

// Install Gity
sudo apt-get update
sudo apt-get install git

// Install cURL 
sudo apt-get install curl

// Install Composer
curl -sS https://getcomposer.org/installer | php

// Make composer.phar global
sudo mv composer.phar /usr/local/bin/composer

// Instal php 5 cURL
sudo apt-get install php5-curl

// Enable mod-rewrite
sudo a2enmod rewrite
sudo service apache2 restart


// Install MCrypt
sudo apt-get install php5-mcrypt

// Enable MCrypt
sudo php5enmod mcrypt

// Goto web directory
cd /var/www

// Allow writing to web directory
sudo chmod -R a+rwx /var/www

// Restart Apache
sudo service apache2 restart

// Lists files in ssh directory, then if needed proceed to generate RSA key
ls -al ~/.ssh  

// Genereates a new key
ssh-keygen -t rsa -C "email@gmail.com" 

// Start the ssh-agent in the background
eval "$(ssh-agent -s)"

// Add it to your keys folder
ssh-add ~/.ssh/id_rsa

// Copy to your server
ssh-copy-id root@yoursite.com

git clone root@yoursite.com:yourprojectname.git 
or
git clone user@123.45.56.78:yourprojectname.git

// Edit your .gitignore file
Add all config files and nbproject folder to .gitignore

// Create new project
composer create-project laravel/laravel projectname --prefer-dist

// Enable writes to app storage directory
chmod -R a+rwx app/storage

// Update Hosts File
sudo vim /etc/apache2/sites-available/000-default.conf 

or 

// *Better, create new file called default.conf
sudo vim /etc/apache2/sites-available/default.conf

Add this
<VirtualHost *:80>
    SetEnv LARAVEL_ENV local
    DocumentRoot "/var/www/yourfolder/public"
    ServerName yoursite.here

    <Directory "/var/www/yourfolder/public">
        Options +FollowSymLinks -MultiViews
        AllowOverride all
    </Directory>
</VirtualHost>

// Update Local Hosts File
sudo vim /etc/hosts

sudo service apache2 restart

composer update

//****** Vim Extensions *********

*******************BEGIN COPY PASTE AREA

" set the runtime path to include Vundle and initialize
set rtp+=~/.vim/bundle/Vundle.vim
call vundle#begin()

Plugin 'gmarik/Vundle.vim'

" All of your Plugins must be added before the following line
call vundle#end() " required
filetype plugin indent on " required
******************END OF COPY PASTE AREA
Then Restart Vim
:BundleInstall

***To Update Plugins

***To remove a Plugin
