# Creating a Ghost blog on a new AWS AMI instance

sudo yum update
wget http://nodejs.org/dist/v0.10.32/node-v0.10.32.tar.gz
tar -zxvf node-v0.10.32.tar.gz
cd node-v0.10.32

sudo yum groupinstall "Development Tools"
./configure
make
sudo make install

sudo mkdir -p /var/www
cd /var/www

sudo wget https://ghost.org/zip/ghost-0.5.2.zip

sudo unzip ghost-0.5.2.zip
